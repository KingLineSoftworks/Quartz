    QUARTZ_RENDERING_Device

    Device.hpp

)

    PUBLIC
    ${QUARTZ_INCLUDE_DIRS}
)

    QUARTZ_RENDERING_Device
    PUBLIC ${QUARTZ_CMAKE_CXX_FLAGS}
)

    QUARTZ_RENDERING_Device
    PUBLIC ${QUARTZ_COMPILE_DEFINITIONS}
)

target_compile_definitions(
        QUARTZ_RENDERING_Device
        PUBLIC ${QUARTZ_COMPILE_DEFINITIONS}
)

target_link_libraries(
    QUARTZ_RENDERING_Device

    PUBLIC
    vulkan

    PUBLIC
    UTIL_Logger

    PUBLIC
    QUARTZ_RENDERING_Instance
)
